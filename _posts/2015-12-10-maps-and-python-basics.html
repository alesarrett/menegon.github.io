Accessing elements of dictionary¶

In [10]:
m.shadedrelief(scale=0.2)

m.scatter(cities['newyork'][0],
          cities['newyork'][1],
          color = 'red',
          latlon=True
         )

m.scatter(cities['rome'][0],
          cities['rome'][1],
          color = 'green',
          latlon=True
         )

m.drawgreatcircle(
    cities['newyork'][0],
    cities['newyork'][1],
    cities['rome'][0],
    cities['rome'][1]
)
